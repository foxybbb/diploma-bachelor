\begin{luacode*}
	function split(inputstr, sep)
	        if sep == nil then
	                sep = "%s"
	        end
	        local t={}
	        for str in string.gmatch(inputstr, "([^"..sep.."]+)") do
	                table.insert(t, str)
	        end
	        return t
	end

	function boldenfirstword(line)
		local words = split(line, " ")
		local fword = words[1]
		table.remove(words, 1)
		tex.print("\\textbf{"..fword.."} "..table.concat(words," "))
	end
\end{luacode*}
